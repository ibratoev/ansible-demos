1. Ping all hosts
ansible all -i hosts -m ping

2. Show patterns:
ansible 'all,!localhost' -i hosts -m ping
ansible 'ub*' -i hosts -m ping

3. Show the setup module on a single machine
ansible ubuntu1 -i hosts -m setup

4. Show the shell module - no Python/PS needed
ansible ubuntu1 -i hosts -m shell -a 'pwd'
ansible all -i hosts -a 'ls'

5. Show the raw module 
ansible ubuntu -i hosts -m raw -a '/usr/bin/cal'
