# Error - needs username and password
ansible all -i hosts_min -m ping

# Use vvvv for more info
ansible all -i hosts_min -m ping -vvvvv

# Error no password
ansible all -i hosts_min -m ping -u vagrant

# Ask for password
ansible all -i hosts_min -m ping -u vagrant -k

# Show complete inventory
ansible all -i hosts -m ping

# Show patterns:
ansible 'all,!localhost' -i hosts -m ping
ansible 'ub*' -i hosts -m ping

# Show the setup module
ansible ubuntu1 -i hosts -m setup

# Show the shell module - no Python/PS needed
ansible ubuntu1 -i hosts -m shell -a 'ls'
ansible all -i hosts -a 'ls'

# Show the raw module 
ansible ubuntu -i hosts -m raw -a '/usr/bin/cal'
